    func checkForMissingFiles(withMinorErrors: Bool = true) {
        let FEX = FileExtractor.shared
        
        var errors = Set<String>()
        var minorErrors = Set<String>()
        
        for statesArray in states {
            
            for state in statesArray {
                
                let Q = state.question
                if Q.isEmpty { continue }
                                
                switch Q.type {
                case .justText:
                    break
                case .textAnsImage:
                    if FEX.getImagePath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing image: \(Q.code + "A")")
                    }
                case .textAnsMusic:
                    if FEX.getImagePath(code: Q.code, roundStatus: .answer) == nil {
                        minorErrors.insert("Images/\(Q.code + "A")")
                    }
                    if FEX.getSoundPath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing sound: \(Q.code + "A")")
                    }
                case .textAnsVideo:
                    if FEX.getVideoPath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing video: \(Q.code + "A")")
                    }
                case .imageAnsImage:
                    if FEX.getImagePath(code: Q.code, roundStatus: .question) == nil {
                        errors.insert("Missing image: \(Q.code + "Q")")
                    }
                    if FEX.getImagePath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing image: \(Q.code + "A")")
                    }
                case .imageAnsMusic:
                    if FEX.getImagePath(code: Q.code, roundStatus: .question) == nil {
                        errors.insert("Missing image: \(Q.code + "Q")")
                    }
                    if FEX.getImagePath(code: Q.code, roundStatus: .answer) == nil {
                        minorErrors.insert("Images/\(Q.code + "A")")
                    }
                    if FEX.getSoundPath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing sound: \(Q.code + "A")")
                    }
                case .imageAnsVideo:
                    if FEX.getImagePath(code: Q.code, roundStatus: .question) == nil {
                        errors.insert("Missing image: \(Q.code + "Q")")
                    }
                    if FEX.getVideoPath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing video: \(Q.code + "A")")
                    }
                case .music:
                    if FEX.getSoundPath(code: Q.code, roundStatus: .question) == nil {
                        errors.insert("Missing sound: \(Q.code + "Q")")
                    }
                    if FEX.getSoundPath(code: Q.code, roundStatus: .rep) == nil {
                        minorErrors.insert("Sounds/\(Q.code + "R")")
                    }
                    if FEX.getSoundPath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing sound: \(Q.code + "A")")
                    }
                case .video:
                    if FEX.getVideoPath(code: Q.code, roundStatus: .question) == nil {
                        errors.insert("Missing video: \(Q.code + "Q")")
                    }
                    if FEX.getVideoPath(code: Q.code, roundStatus: .rep) == nil {
                        minorErrors.insert("Videos/\(Q.code + "R")")
                    }
                    if FEX.getVideoPath(code: Q.code, roundStatus: .answer) == nil {
                        errors.insert("Missing video: \(Q.code + "A")")
                    }
                }
            }
        }
        
        errors = errors.filter { errorString in
            !missingExceptions.contains { errorString.hasSuffix($0) }
        }
        minorErrors = minorErrors.filter { errorString in
            !missingExceptions.contains { errorString.hasSuffix($0) }
        }
        
        for error in errors.sorted() {
            print(error)
        }
        if withMinorErrors && !minorErrors.isEmpty {
            print("\nMinor errors:")
            for error in minorErrors.sorted() {
                print(error)
            }
        }
    }
